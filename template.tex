    S = y(t,1); E = y(t,2); I = y(t,3); R = y(t,4); D = y(t,5);

    dS = -beta * S * I;
    dE = beta * S * I - sigma * E;
    dI = sigma * E - gamma * I - mu * I;
    dR = gamma * I;
    dD = mu * I;

    y(t+1,:) = y(t,:) + dt * [dS, dE, dI, dR, dD];
